3//! SKY130 PDK behavioral model parser and AIG decomposition.
4//!
5//! Parses gate-level functional Verilog models from the official SKY130 PDK
6//! (google/skywater-pdk-libs-sky130_fd_sc_hd) and converts them to AIG
7//! decompositions. This replaces hand-coded decompositions with vendor-verified
8//! reference models.
9//!
10//! The functional models use only Verilog gate primitives:
11//!   and, or, nand, nor, not, xor, xnor, buf
12//! Some cells (mux2, mux2i, mux4) use UDP truth tables instead.
13

132/// Result of decomposing a cell into AIG operations.
133///
134/// The decomposition produces a sequence of AND gates that must be built
135/// in order, where later gates can reference earlier ones.
136#[derive(Debug, Clone)]
137pub struct DecompResult {
138    /// Sequence of AND gate operations to build.
139    /// Each entry is (input_a_iv, input_b_iv) where the lower bit is inversion.
140    /// References to earlier gates use negative indices (-1 = first gate output, etc.)
141    pub and_gates: Vec<(i64, i64)>,
142    /// Index of the final output (-1 = first gate, -2 = second gate, etc.)
143    /// Positive values reference original inputs.
144    pub output_idx: i64,
145    /// Whether to invert the final output
146    pub output_inverted: bool,
147}

149/// Check if a cell type is a sequential element (DFF or latch).
150///
151/// This is the exhaustive list of sky130_fd_sc_hd cells containing DFF or
152/// latch UDPs in their behavioral Verilog models. Derived from the PDK by
153/// grepping for `udp_dff` and `udp_dlatch` primitives in behavioral.v files.
154///
155/// IMPORTANT: Do NOT use prefix matching here â€” `dlygate*` and `dlymetal*`
156/// are combinational delay buffers that happen to start with "dl".
157const SKY130_SEQUENTIAL_CELLS: &[&str] = &[
158    // D flip-flops
159    "dfbbn", "dfbbp", "dfrbp", "dfrtn", "dfrtp", "dfsbp", "dfstp", "dfxbp", "dfxtp",
160    // Latches and clock-gating latches
161    "dlclkp", "dlrbn", "dlrbp", "dlrtn", "dlrtp", "dlxbn", "dlxbp", "dlxtn", "dlxtp",
162    // Enable D flip-flops
163    "edfxbp", "edfxtp",
164    // Low-power isolation latch
165    "lpflow_inputisolatch",
166    // Scan D flip-flops
167    "sdfbbn", "sdfbbp", "sdfrbp", "sdfrtn", "sdfrtp", "sdfsbp", "sdfstp", "sdfxbp", "sdfxtp",
168    // Scan clock-gating latch
169    "sdlclkp",
170    // Scan enable D flip-flops
171    "sedfxbp", "sedfxtp",
172];
173
174pub fn is_sequential_cell(cell_type: &str) -> bool {
175    SKY130_SEQUENTIAL_CELLS.contains(&cell_type)
176}

251/// Parse a functional Verilog model file (*.functional.v).
252///
253/// Handles:
254/// - module declaration with port names and directions
255/// - Gate instantiations: `gate_type name (output, input1, input2, ...);`
256/// - Skips: comments, `supply`, `wire`, `timescale`, `celldefine`, etc.
257pub fn parse_functional_model(verilog_src: &str) -> Option<BehavioralModel> {
258    let mut module_name = String::new();
259    let mut port_names: Vec<String> = Vec::new();
260    let mut input_ports: Vec<String> = Vec::new();
261    let mut output_ports: Vec<String> = Vec::new();
262    let mut gates: Vec<BehavioralGate> = Vec::new();
263
264    // Simple state machine
265    enum State {
266        LookingForModule,
267        InModulePorts,
268        InModuleBody,
269    }
270    let mut state = State::LookingForModule;
271
272    // Accumulate multi-line tokens
273    let mut accum = String::new();
274
275    for line in verilog_src.lines() {
276        let trimmed = line.trim();
277
278        // Skip comments and preprocessor directives
279        if trimmed.starts_with("//")
280            || trimmed.starts_with("/*")
281            || trimmed.starts_with("*/")
282            || trimmed.starts_with('*')
283            || trimmed.starts_with('`')
284            || trimmed.is_empty()
285        {
286            continue;
287        }
288
289        match state {
290            State::LookingForModule => {
291                if trimmed.starts_with("module ") {
292                    // Parse: module sky130_fd_sc_hd__o21ai (
293                    accum.clear();
294                    accum.push_str(trimmed);
295
296                    if accum.contains(");") {
297                        parse_module_header(&accum, &mut module_name, &mut port_names);
298                        state = State::InModuleBody;
299                    } else {
300                        state = State::InModulePorts;
301                    }
302                }
303            }
304            State::InModulePorts => {
305                accum.push(' ');
306                accum.push_str(trimmed);
307                if accum.contains(");") {
308                    parse_module_header(&accum, &mut module_name, &mut port_names);
309                    state = State::InModuleBody;
310                }
311            }
312            State::InModuleBody => {
313                if trimmed == "endmodule" {
314                    break;
315                }
316
317                // Parse port direction declarations
318                if trimmed.starts_with("output ") {
319                    let names = parse_port_declaration(trimmed, "output");
320                    output_ports.extend(names);
321                    continue;
322                }
323                if trimmed.starts_with("input ") {
324                    let names = parse_port_declaration(trimmed, "input");
325                    input_ports.extend(names);
326                    continue;
327                }
328
329                // Skip non-gate lines
330                if trimmed.starts_with("wire ")
331                    || trimmed.starts_with("supply")
332                    || trimmed.starts_with("pullup")
333                    || trimmed.starts_with("pulldown")
334                    || trimmed.starts_with("reg ")
335                {
336                    continue;
337                }
338
339                // Try to parse gate instantiation
340                // Could be multi-line, accumulate until we see ';'
341                if !trimmed.starts_with("//") && !trimmed.is_empty() {
342                    accum.clear();
343                    accum.push_str(trimmed);
344                    if !accum.contains(';') {
345                        // Multi-line gate instantiation - keep accumulating
346                        // (unusual in functional models but handle it)
347                        continue;
348                    }
349                    if let Some(gate) = parse_gate_instantiation(&accum) {
350                        gates.push(gate);
351                    }
352                }
353            }
354        }
355    }
356
357    if module_name.is_empty() {
358        return None;
359    }
360
361    Some(BehavioralModel {
362        module_name,
363        inputs: input_ports,
364        outputs: output_ports,
365        gates,
366    })
367}

369/// Parse module header: `module name ( port1, port2, ... );`
370fn parse_module_header(header: &str, name: &mut String, ports: &mut Vec<String>) {
371    // Find module name
372    let after_module = header.strip_prefix("module ").unwrap_or(header);
373    if let Some(paren_pos) = after_module.find('(') {
374        *name = after_module[..paren_pos].trim().to_string();
375
376        // Extract ports between ( and )
377        let rest = &after_module[paren_pos + 1..];
378        if let Some(close) = rest.find(')') {
379            let port_str = &rest[..close];
380            *ports = port_str
381                .split(',')
382                .map(|s| s.trim().to_string())
383                .filter(|s| !s.is_empty())
384                .collect();
385        }
386    }
387}

389/// Parse port declarations like `output Y ;` or `input A1;`
390fn parse_port_declaration(line: &str, keyword: &str) -> Vec<String> {
391    let after_kw = line
392        .strip_prefix(keyword)
393        .unwrap_or(line)
394        .trim()
395        .trim_end_matches(';')
396        .trim();
397    after_kw
398        .split(',')
399        .map(|s| s.trim().to_string())
400        .filter(|s| !s.is_empty())
401        .collect()
402}

404/// Parse a gate instantiation line like:
405/// `or   or0   (or0_out    , A2, A1         );`
406/// Returns BehavioralGate or None if not a gate.
407fn parse_gate_instantiation(line: &str) -> Option<BehavioralGate> {
408    let line = line.trim().trim_end_matches(';').trim();
409
410    // Find the parenthesized port list
411    let paren_start = line.find('(')?;
412    let paren_end = line.rfind(')')?;
413
414    // Everything before '(' is: gate_type [#delay] instance_name
415    let prefix = line[..paren_start].trim();
416    let port_list = &line[paren_start + 1..paren_end];
417
418    // Split prefix into tokens
419    let tokens: Vec<&str> = prefix.split_whitespace().collect();
420    if tokens.is_empty() {
421        return None;
422    }
423
424    let gate_type = tokens[0].to_string();
425
426    // Skip delay specifications like `UNIT_DELAY
427    // (used in DFF models - we skip those cells entirely)
428
429    // Parse port list: output is first, then inputs
430    let ports: Vec<String> = port_list
431        .split(',')
432        .map(|s| s.trim().to_string())
433        .filter(|s| !s.is_empty())
434        .collect();
435
436    if ports.is_empty() {
437        return None;
438    }
439
440    // Skip named port connections (module instantiations like .A(wire))
441    // These are hierarchical cell references, not gate primitives
442    if ports[0].starts_with('.') {
443        return None;
444    }
445
446    let output = ports[0].clone();
447    let inputs = ports[1..].to_vec();
448
449    Some(BehavioralGate {
450        gate_type,
451        output,
452        inputs,
453    })
454}

460/// Parse a Verilog UDP primitive definition.
461pub fn parse_udp(verilog_src: &str) -> Option<UdpModel> {
462    let mut name = String::new();
463    let mut output_port = String::new();
464    let mut input_ports: Vec<String> = Vec::new();
465    let mut rows: Vec<UdpRow> = Vec::new();
466
467    enum State {
468        LookingForPrimitive,
469        InPrimitivePorts,
470        InPrimitiveBody,
471        InTable,
472    }
473    let mut state = State::LookingForPrimitive;
474    let mut accum = String::new();
475
476    for line in verilog_src.lines() {
477        let trimmed = line.trim();
478
479        if trimmed.starts_with("//")
480            || trimmed.starts_with("/*")
481            || trimmed.starts_with("*/")
482            || trimmed.starts_with('*')
483            || trimmed.starts_with('`')
484            || trimmed.is_empty()
485        {
486            continue;
487        }
488
489        match state {
490            State::LookingForPrimitive => {
491                if trimmed.starts_with("primitive ") {
492                    accum.clear();
493                    accum.push_str(trimmed);
494                    if accum.contains(");") {
495                        parse_primitive_header(&accum, &mut name);
496                        state = State::InPrimitiveBody;
497                    } else {
498                        state = State::InPrimitivePorts;
499                    }
500                }
501            }
502            State::InPrimitivePorts => {
503                accum.push(' ');
504                accum.push_str(trimmed);
505                if accum.contains(");") {
506                    parse_primitive_header(&accum, &mut name);
507                    state = State::InPrimitiveBody;
508                }
509            }
510            State::InPrimitiveBody => {
511                if trimmed == "endprimitive" {
512                    break;
513                }
514                if trimmed.starts_with("output ") {
515                    output_port = trimmed
516                        .strip_prefix("output ")
517                        .unwrap()
518                        .trim()
519                        .trim_end_matches(';')
520                        .trim()
521                        .to_string();
522                } else if trimmed.starts_with("input ") {
523                    let ports_str = trimmed
524                        .strip_prefix("input ")
525                        .unwrap()
526                        .trim()
527                        .trim_end_matches(';')
528                        .trim();
529                    // Accumulate input ports (may be on separate lines)
530                    for port in ports_str.split(',') {
531                        let p = port.trim().to_string();
532                        if !p.is_empty() {
533                            input_ports.push(p);
534                        }
535                    }
536                } else if trimmed == "table" {
537                    state = State::InTable;
538                }
539            }
540            State::InTable => {
541                if trimmed == "endtable" {
542                    state = State::InPrimitiveBody;
543                    continue;
544                }
545
546                // Parse table row like: `0   ?   0  :  1   ;`
547                let row_str = trimmed.trim_end_matches(';').trim();
548                // Remove comment prefix (//  A0  A1  S  :  Y)
549                if row_str.starts_with("//") {
550                    continue;
551                }
552
553                if let Some(colon_pos) = row_str.find(':') {
554                    let input_str = row_str[..colon_pos].trim();
555                    let output_str = row_str[colon_pos + 1..].trim();
556
557                    let input_vals: Vec<Option<bool>> = input_str
558                        .split_whitespace()
559                        .map(|s| match s {
560                            "0" => Some(false),
561                            "1" => Some(true),
562                            "?" | "x" | "X" => None,
563                            _ => None,
564                        })
565                        .collect();
566
567                    let output_val = match output_str {
568                        "1" => true,
569                        "0" => false,
570                        _ => continue, // skip 'x' outputs
571                    };
572
573                    if input_vals.len() == input_ports.len() {
574                        rows.push(UdpRow {
575                            inputs: input_vals,
576                            output: output_val,
577                        });
578                    }
579                }
580            }
581        }
582    }
583
584    if name.is_empty() {
585        return None;
586    }
587
588    Some(UdpModel {
589        name,
590        output: output_port,
591        inputs: input_ports,
592        rows,
593    })
594}

607/// Evaluate a UDP truth table for specific input values.
608/// Returns the output value. Panics if no matching row found.
609fn eval_udp_for_inputs(udp: &UdpModel, input_vals: &[bool]) -> bool {
610    assert_eq!(input_vals.len(), udp.inputs.len());
611
612    for row in &udp.rows {
613        let matches = row
614            .inputs
615            .iter()
616            .zip(input_vals.iter())
617            .all(|(pattern, &actual)| match pattern {
618                None => true,            // don't-care matches anything
619                Some(v) => *v == actual, // must match exactly
620            });
621        if matches {
622            return row.output;
623        }
624    }
625
626    panic!(
627        "UDP '{}': no matching row for inputs {:?}",
628        udp.name, input_vals
629    );
630}

632/// Internal wire value during AIG construction from a behavioral model.
633/// Can be either a reference to a module input (by aigpin_iv) or an intermediate
634/// AND gate we built (by gate index).
635#[derive(Clone, Copy, Debug)]
636enum WireVal {
637    /// An AIG pin with inversion bit (aigpin_iv). Bit 0 = inverted.
638    AigPin(usize),
639    /// Constant value
640    Const(bool),
641}
642
643impl WireVal {
644    /// Get the aigpin_iv value, creating const-0 = AigPin(0) convention.
645    fn as_aigpin_iv(self) -> i64 {
646        match self {
647            WireVal::AigPin(iv) => iv as i64,
648            WireVal::Const(false) => 0, // const-0
649            WireVal::Const(true) => 1,  // const-1
650        }
651    }
652
653    /// Invert this wire value.
654    fn inverted(self) -> Self {
655        match self {
656            WireVal::AigPin(iv) => WireVal::AigPin(iv ^ 1),
657            WireVal::Const(v) => WireVal::Const(!v),
658        }
659    }
660}

662/// Convert a parsed behavioral model to an AIG decomposition for a specific output pin.
663///
664/// # Arguments
665/// * `model` - The parsed behavioral model
666/// * `cell_inputs` - CellInputs with aigpin_iv values for each module input
667/// * `output_pin` - Which output pin to build the logic cone for (e.g. "Y", "SUM", "COUT")
668/// * `udps` - Available UDP models for resolving UDP instantiations
669///
670/// # Returns
671/// A `DecompResult` describing the AIG for this output
672pub fn decompose_from_behavioral(
673    model: &BehavioralModel,
674    cell_inputs: &CellInputs,
675    output_pin: &str,
676    udps: &HashMap<String, UdpModel>,
677) -> DecompResult {
678    // Map wire names to their values
679    let mut wires: HashMap<String, WireVal> = HashMap::new();
680
681    // Map module input port names to their aigpin_iv values from CellInputs
682    for input_name in &model.inputs {
683        let aigpin_iv = get_cell_input_by_name(cell_inputs, input_name);
684        assert_ne!(
685            aigpin_iv,
686            usize::MAX,
687            "Input pin '{}' not set in CellInputs for cell '{}'",
688            input_name,
689            model.module_name
690        );
691        wires.insert(input_name.clone(), WireVal::AigPin(aigpin_iv));
692    }
693
694    // Process gates in order (they're topologically sorted in the Verilog)
695    let mut and_gates: Vec<(i64, i64)> = Vec::new();
696    // Map from gate output index to AND gate index in and_gates
697    // gate_idx_for_wire[wire_name] = index into and_gates (0-based)
698
699    for gate in &model.gates {
700        let gate_type = gate.gate_type.as_str();
701
702        // Skip buf gates - they're just passthroughs
703        if gate_type == "buf" {
704            assert_eq!(gate.inputs.len(), 1, "buf must have exactly 1 input");
705            let input_val = wires
706                .get(&gate.inputs[0])
707                .copied()
708                .unwrap_or_else(|| panic!("Unknown wire '{}' in buf gate", gate.inputs[0]));
709            wires.insert(gate.output.clone(), input_val);
710            continue;
711        }
712
713        // Handle UDP instantiations
714        if gate_type.starts_with("sky130_fd_sc_hd__udp_") {
715            let result = build_udp_aig(gate, &wires, udps, &mut and_gates);
716            wires.insert(gate.output.clone(), result);
717            continue;
718        }
719
720        // Resolve all input wires
721        let input_vals: Vec<WireVal> = gate
722            .inputs
723            .iter()
724            .map(|name| {
725                wires
726                    .get(name)
727                    .copied()
728                    .unwrap_or_else(|| panic!("Unknown wire '{}' in {} gate", name, gate_type))
729            })
730            .collect();
731
732        let result = match gate_type {
733            "not" => {
734                assert_eq!(input_vals.len(), 1);
735                input_vals[0].inverted()
736            }
737            "and" => build_chain_gate(&input_vals, false, false, &mut and_gates),
738            "nand" => build_chain_gate(&input_vals, false, true, &mut and_gates),
739            "or" => build_chain_gate(&input_vals, true, true, &mut and_gates),
740            "nor" => build_chain_gate(&input_vals, true, false, &mut and_gates),
741            "xor" => build_xor_chain(&input_vals, false, &mut and_gates),
742            "xnor" => build_xor_chain(&input_vals, true, &mut and_gates),
743            _ => panic!(
744                "Unknown gate type '{}' in model '{}'",
745                gate_type, model.module_name
746            ),
747        };
748
749        wires.insert(gate.output.clone(), result);
750    }
751
752    // Get the output wire value
753    let output_val = wires
754        .get(output_pin)
755        .copied()
756        .unwrap_or_else(|| panic!("Output pin '{}' not found in model", output_pin));
757
758    // Convert to DecompResult
759    match output_val {
760        WireVal::AigPin(iv) => {
761            let pin_idx = iv >> 1;
762            let inverted = (iv & 1) != 0;
763            DecompResult {
764                and_gates,
765                output_idx: pin_idx as i64,
766                output_inverted: inverted,
767            }
768        }
769        WireVal::Const(v) => {
770            // Constant output: use pin index 0 which is const-0
771            DecompResult {
772                and_gates,
773                output_idx: 0,
774                output_inverted: v,
775            }
776        }
777    }
778}

780/// Build an AND/NAND/OR/NOR chain for multi-input gates.
781///
782/// For AND/NAND: compute AND of all inputs, optionally invert at the end.
783/// For OR/NOR: invert all inputs, AND them, optionally invert at the end.
784///   OR(a,b,c) = NOT(AND(NOT a, NOT b, NOT c))
785///   NOR(a,b,c) = AND(NOT a, NOT b, NOT c)
786fn build_chain_gate(
787    inputs: &[WireVal],
788    invert_inputs: bool,
789    invert_output: bool,
790    and_gates: &mut Vec<(i64, i64)>,
791) -> WireVal {
792    assert!(inputs.len() >= 2, "Gate must have at least 2 inputs");
793
794    let inputs: Vec<WireVal> = if invert_inputs {
795        inputs.iter().map(|v| v.inverted()).collect()
796    } else {
797        inputs.to_vec()
798    };
799
800    // Chain 2-input AND gates
801    let mut accum = inputs[0];
802    for input in &inputs[1..] {
803        let a_ref = accum.as_aigpin_iv();
804        let b_ref = input.as_aigpin_iv();
805        and_gates.push((a_ref, b_ref));
806        let gate_idx = and_gates.len() - 1;
807        accum = WireVal::AigPin(GATE_MARKER | (gate_idx << 1));
808    }
809
810    if invert_output {
811        accum.inverted()
812    } else {
813        accum
814    }
815}
816
817/// Marker bit to distinguish gate references from pin references.
818/// Gate outputs use bit 30 set. This limits us to ~500M gates (more than enough).
819const GATE_MARKER: usize = 1 << 30;
820
821/// Check if an aigpin_iv value is a gate reference.
822fn is_gate_ref(aigpin_iv: usize) -> bool {
823    aigpin_iv & GATE_MARKER != 0
824}
825
826/// Extract gate index from a gate-reference aigpin_iv.
827fn gate_ref_index(aigpin_iv: usize) -> usize {
828    (aigpin_iv & !GATE_MARKER & !1) >> 1
829}

831/// Build a 2-input XOR: A ^ B = !(!( A & !B) & !(!A & B))
832fn build_xor_2(a: WireVal, b: WireVal, and_gates: &mut Vec<(i64, i64)>) -> WireVal {
833    let a_iv = a.as_aigpin_iv();
834    let b_iv = b.as_aigpin_iv();
835    let a_inv_iv = a.inverted().as_aigpin_iv();
836    let b_inv_iv = b.inverted().as_aigpin_iv();
837
838    // gate0: A & !B
839    and_gates.push((a_iv, b_inv_iv));
840    let g0 = and_gates.len() - 1;
841    let g0_val = WireVal::AigPin(GATE_MARKER | (g0 << 1));
842
843    // gate1: !A & B
844    and_gates.push((a_inv_iv, b_iv));
845    let g1 = and_gates.len() - 1;
846    let g1_val = WireVal::AigPin(GATE_MARKER | (g1 << 1));
847
848    // gate2: !(A & !B) & !(!A & B)  -- this is XNOR, inverted gives XOR
849    let g0_inv_iv = g0_val.inverted().as_aigpin_iv();
850    let g1_inv_iv = g1_val.inverted().as_aigpin_iv();
851    and_gates.push((g0_inv_iv, g1_inv_iv));
852    let g2 = and_gates.len() - 1;
853    // XOR = NOT(gate2), so return inverted
854    WireVal::AigPin(GATE_MARKER | (g2 << 1) | 1)
855}
856
857/// Build XOR/XNOR chain for multi-input gates.
858fn build_xor_chain(
859    inputs: &[WireVal],
860    invert_output: bool,
861    and_gates: &mut Vec<(i64, i64)>,
862) -> WireVal {
863    assert!(inputs.len() >= 2);
864
865    let mut accum = inputs[0];
866    for input in &inputs[1..] {
867        accum = build_xor_2(accum, *input, and_gates);
868    }
869
870    if invert_output {
871        accum.inverted()
872    } else {
873        accum
874    }
875}

877/// Build AIG for a UDP instantiation by converting truth table to sum-of-products.
878fn build_udp_aig(
879    gate: &BehavioralGate,
880    wires: &HashMap<String, WireVal>,
881    udps: &HashMap<String, UdpModel>,
882    and_gates: &mut Vec<(i64, i64)>,
883) -> WireVal {
884    let udp_name = &gate.gate_type;
885    let udp = udps
886        .get(udp_name)
887        .unwrap_or_else(|| panic!("UDP '{}' not found in loaded models", udp_name));
888
889    // Get input wire values
890    let input_vals: Vec<WireVal> = gate
891        .inputs
892        .iter()
893        .map(|name| {
894            wires
895                .get(name)
896                .copied()
897                .unwrap_or_else(|| panic!("Unknown wire '{}' in UDP '{}'", name, udp_name))
898        })
899        .collect();
900
901    assert_eq!(
902        input_vals.len(),
903        udp.inputs.len(),
904        "UDP '{}' expects {} inputs, got {}",
905        udp_name,
906        udp.inputs.len(),
907        input_vals.len()
908    );
909
910    // Build sum-of-products from truth table rows where output=1
911    // Each row with output=1 becomes a product (AND) term.
912    // Product terms are ORed together.
913    //
914    // For rows where output=0, we don't need to do anything explicitly.
915    // Don't-care (?) inputs are omitted from the product term.
916
917    let one_rows: Vec<&UdpRow> = udp.rows.iter().filter(|r| r.output).collect();
918
919    if one_rows.is_empty() {
920        // Output is always 0
921        return WireVal::Const(false);
922    }
923
924    // Build each product term
925    let mut product_terms: Vec<WireVal> = Vec::new();
926
927    for row in &one_rows {
928        // Collect non-don't-care inputs for this product term
929        let mut term_inputs: Vec<WireVal> = Vec::new();
930        for (i, pattern) in row.inputs.iter().enumerate() {
931            match pattern {
932                Some(true) => term_inputs.push(input_vals[i]),
933                Some(false) => term_inputs.push(input_vals[i].inverted()),
934                None => {} // don't-care - omit from product
935            }
936        }
937
938        if term_inputs.is_empty() {
939            // All inputs are don't-care: output is unconditionally 1
940            return WireVal::Const(true);
941        }
942
943        if term_inputs.len() == 1 {
944            product_terms.push(term_inputs[0]);
945        } else {
946            // Build AND chain for this product term
947            let product = build_chain_gate(&term_inputs, false, false, and_gates);
948            product_terms.push(product);
949        }
950    }
951
952    if product_terms.len() == 1 {
953        return product_terms[0];
954    }
955
956    // OR the product terms: OR(a,b,...) = NOT(AND(NOT a, NOT b, ...))
957    build_chain_gate(&product_terms, true, true, and_gates)
958}

1000/// Post-process a DecompResult built with GATE_MARKER encoding to use
1001/// standard negative-index encoding for the and_gates references.
1002fn finalize_decomp_result(and_gates: Vec<(i64, i64)>, output: WireVal) -> DecompResult {
1003    // Convert gate references in and_gates from GATE_MARKER to negative indices
1004    let converted_gates: Vec<(i64, i64)> = and_gates
1005        .iter()
1006        .map(|(a, b)| (convert_ref_to_standard(*a), convert_ref_to_standard(*b)))
1007        .collect();
1008
1009    match output {
1010        WireVal::AigPin(iv) if is_gate_ref(iv) => {
1011            let gate_idx = gate_ref_index(iv);
1012            let inverted = (iv & 1) != 0;
1013            DecompResult {
1014                and_gates: converted_gates,
1015                output_idx: -(gate_idx as i64) - 1,
1016                output_inverted: inverted,
1017            }
1018        }
1019        WireVal::AigPin(iv) => {
1020            let pin_idx = iv >> 1;
1021            let inverted = (iv & 1) != 0;
1022            DecompResult {
1023                and_gates: converted_gates,
1024                output_idx: pin_idx as i64,
1025                output_inverted: inverted,
1026            }
1027        }
1028        WireVal::Const(v) => DecompResult {
1029            and_gates: converted_gates,
1030            output_idx: 0,
1031            output_inverted: v,
1032        },
1033    }
1034}
1035
1036/// Convert a single reference value from GATE_MARKER encoding to standard.
1037fn convert_ref_to_standard(ref_val: i64) -> i64 {
1038    let uval = ref_val as usize;
1039    if is_gate_ref(uval) {
1040        let gate_idx = gate_ref_index(uval);
1041        let inverted = (uval & 1) != 0;
1042        let base = -((gate_idx as i64) * 2 + 1);
1043        if inverted {
1044            base ^ 1
1045        } else {
1046            base
1047        }
1048    } else {
1049        ref_val
1050    }
1051}

1057/// Validate that all gate inputs in a model reference known wires.
1058/// Returns false if any gate references a wire that isn't a module input
1059/// or the output of a previous gate.
1060fn validate_model(model: &BehavioralModel) -> bool {
1061    let mut known_wires: std::collections::HashSet<&str> = std::collections::HashSet::new();
1062
1063    // Module inputs are known
1064    for name in &model.inputs {
1065        known_wires.insert(name.as_str());
1066    }
1067
1068    for gate in &model.gates {
1069        // Check all inputs are known
1070        for input_name in &gate.inputs {
1071            if !known_wires.contains(input_name.as_str()) {
1072                return false;
1073            }
1074        }
1075        // Gate output becomes known
1076        known_wires.insert(gate.output.as_str());
1077    }
1078
1079    // Check that all declared outputs are produced
1080    for output_name in &model.outputs {
1081        if !known_wires.contains(output_name.as_str()) {
1082            return false;
1083        }
1084    }
1085
1086    true
1087}

1089/// Load all PDK models needed for a set of cell types.
1090///
1091/// # Arguments
1092/// * `pdk_cells_path` - Path to the `cells/` directory of the PDK
1093/// * `cell_types` - Set of cell type names that need models (e.g. ["o21ai", "nand2", ...])
1094///
1095/// # Returns
1096/// A PdkModels containing all loaded models and UDPs.
1097pub fn load_pdk_models(pdk_cells_path: &Path, cell_types: &[String]) -> PdkModels {
1098    let mut models = HashMap::new();
1099    let mut udps: HashMap<String, UdpModel> = HashMap::new();
1100
1101    // Derive models_path from cells_path: ../models/
1102    let models_path = pdk_cells_path.parent().unwrap().join("models");
1103
1104    for cell_type in cell_types {
1105        // Skip sequential and tie cells - handled elsewhere
1106        if is_sequential_cell(cell_type) || is_tie_cell(cell_type) {
1107            continue;
1108        }
1109
1110        let cell_dir = pdk_cells_path.join(cell_type);
1111        let func_file = cell_dir.join(format!("sky130_fd_sc_hd__{}.functional.v", cell_type));
1112
1113        if !func_file.exists() {
1114            clilog::warn!(
1115                "PDK functional model not found for cell '{}': {}",
1116                cell_type,
1117                func_file.display()
1118            );
1119            continue;
1120        }
1121
1122        let src = std::fs::read_to_string(&func_file)
1123            .unwrap_or_else(|e| panic!("Failed to read PDK model {}: {}", func_file.display(), e));
1124
1125        if let Some(model) = parse_functional_model(&src) {
1126            // Validate: check that all gate inputs reference known wires
1127            // (module inputs, or outputs of earlier gates)
1128            let valid = validate_model(&model);
1129            if !valid {
1130                clilog::debug!(
1131                    "Skipping PDK model '{}': has unresolvable wire references (macro cell?)",
1132                    cell_type
1133                );
1134                continue;
1135            }
1136
1137            // Check if the model uses any UDPs and load them
1138            for gate in &model.gates {
1139                if gate.gate_type.starts_with("sky130_fd_sc_hd__udp_") {
1140                    let udp_name = &gate.gate_type;
1141                    if !udps.contains_key(udp_name) {
1142                        if let Some(udp) = load_udp(&models_path, udp_name) {
1143                            udps.insert(udp_name.clone(), udp);
1144                        }
1145                    }
1146                }
1147            }
1148            models.insert(cell_type.clone(), model);
1149        } else {
1150            clilog::warn!(
1151                "Failed to parse PDK functional model for cell '{}'",
1152                cell_type
1153            );
1154        }
1155    }
1156
1157    clilog::info!(
1158        "Loaded {} PDK cell models and {} UDP models",
1159        models.len(),
1160        udps.len()
1161    );
1162
1163    PdkModels { models, udps }
1164}

1166/// Load a single UDP model from the models directory.
1167fn load_udp(models_path: &Path, udp_name: &str) -> Option<UdpModel> {
1168    // UDP name format: sky130_fd_sc_hd__udp_mux_2to1_N
1169    // Directory format: udp_mux_2to1_n (lowercase)
1170    // File: sky130_fd_sc_hd__udp_mux_2to1_n.v (note: lowercase in filename sometimes differs)
1171
1172    // Extract the udp part after "sky130_fd_sc_hd__"
1173    let udp_suffix = udp_name
1174        .strip_prefix("sky130_fd_sc_hd__")
1175        .unwrap_or(udp_name);
1176
1177    // The directory name is lowercase version of the suffix
1178    let dir_name = udp_suffix.to_lowercase();
1179    let udp_dir = models_path.join(&dir_name);
1180
1181    if !udp_dir.exists() {
1182        clilog::warn!("UDP directory not found: {}", udp_dir.display());
1183        return None;
1184    }
1185
1186    // Find the .v file (not .blackbox.v or .tb.v)
1187    let udp_file = udp_dir.join(format!("sky130_fd_sc_hd__{}.v", dir_name));
1188
1189    if !udp_file.exists() {
1190        clilog::warn!("UDP file not found: {}", udp_file.display());
1191        return None;
1192    }
1193
1194    let src = std::fs::read_to_string(&udp_file)
1195        .unwrap_or_else(|e| panic!("Failed to read UDP file {}: {}", udp_file.display(), e));
1196
1197    parse_udp(&src)
1198}

1200/// Decompose a cell using PDK models.
1201///
1202/// This is the main entry point for PDK-based decomposition. Panics if no
1203/// PDK model is available for the given cell type.
1204pub fn decompose_with_pdk(
1205    cell_type: &str,
1206    inputs: &CellInputs,
1207    output_pin: &str,
1208    pdk: &PdkModels,
1209) -> DecompResult {
1210    let model = pdk.models.get(cell_type).unwrap_or_else(|| {
1211        panic!(
1212            "No PDK model found for cell type '{}'. Ensure the sky130_fd_sc_hd submodule is \
1213             initialized (git submodule update --init) and the cell model exists.",
1214            cell_type
1215        )
1216    });
1217
1218    let mut and_gates = Vec::new();
1219    let mut wires: HashMap<String, WireVal> = HashMap::new();
1220
1221    // Map module input port names to their aigpin_iv values
1222    for input_name in &model.inputs {
1223        let aigpin_iv = get_cell_input_by_name(inputs, input_name);
1224        if aigpin_iv == usize::MAX {
1225            panic!(
1226                "Input pin '{}' not set in CellInputs for cell type '{}'",
1227                input_name, cell_type
1228            );
1229        }
1230        wires.insert(input_name.clone(), WireVal::AigPin(aigpin_iv));
1231    }
1232
1233    // Process gates in order
1234    for gate in &model.gates {
1235        let gate_type = gate.gate_type.as_str();
1236
1237        if gate_type == "buf" {
1238            assert_eq!(gate.inputs.len(), 1);
1239            let input_val = wires
1240                .get(&gate.inputs[0])
1241                .copied()
1242                .unwrap_or_else(|| panic!("Unknown wire '{}' in buf gate", gate.inputs[0]));
1243            wires.insert(gate.output.clone(), input_val);
1244            continue;
1245        }
1246
1247        if gate_type.starts_with("sky130_fd_sc_hd__udp_") {
1248            let result = build_udp_aig(gate, &wires, &pdk.udps, &mut and_gates);
1249            wires.insert(gate.output.clone(), result);
1250            continue;
1251        }
1252
1253        let input_vals: Vec<WireVal> = gate
1254            .inputs
1255            .iter()
1256            .map(|name| {
1257                wires
1258                    .get(name)
1259                    .copied()
1260                    .unwrap_or_else(|| panic!("Unknown wire '{}' in {} gate", name, gate_type))
1261            })
1262            .collect();
1263
1264        let result = match gate_type {
1265            "not" => {
1266                assert_eq!(input_vals.len(), 1);
1267                input_vals[0].inverted()
1268            }
1269            "and" => build_chain_gate(&input_vals, false, false, &mut and_gates),
1270            "nand" => build_chain_gate(&input_vals, false, true, &mut and_gates),
1271            "or" => build_chain_gate(&input_vals, true, true, &mut and_gates),
1272            "nor" => build_chain_gate(&input_vals, true, false, &mut and_gates),
1273            "xor" => build_xor_chain(&input_vals, false, &mut and_gates),
1274            "xnor" => build_xor_chain(&input_vals, true, &mut and_gates),
1275            _ => panic!(
1276                "Unknown gate type '{}' in model '{}'",
1277                gate_type, model.module_name
1278            ),
1279        };
1280
1281        wires.insert(gate.output.clone(), result);
1282    }
1283
1284    // Get the requested output
1285    let output_val = wires.get(output_pin).copied().unwrap_or_else(|| {
1286        panic!(
1287            "Output pin '{}' not found in model '{}'. Available: {:?}",
1288            output_pin, model.module_name, model.outputs
1289        )
1290    });
1291
1292    finalize_decomp_result(and_gates, output_val)
1293}

1299/// Directly evaluate a behavioral model's gate network for given input values.
1300/// This doesn't go through AIG - it directly interprets the Verilog gates.
1301/// Used as a reference oracle in tests.
1302pub fn eval_behavioral_model(
1303    model: &BehavioralModel,
1304    input_values: &HashMap<String, bool>,
1305    output_pin: &str,
1306    udps: &HashMap<String, UdpModel>,
1307) -> bool {
1308    let mut wires: HashMap<String, bool> = HashMap::new();
1309
1310    // Set input values
1311    for (name, &val) in input_values {
1312        wires.insert(name.clone(), val);
1313    }
1314
1315    // Evaluate gates in order
1316    for gate in &model.gates {
1317        let gate_type = gate.gate_type.as_str();
1318
1319        if gate_type == "buf" {
1320            let v = wires[&gate.inputs[0]];
1321            wires.insert(gate.output.clone(), v);
1322            continue;
1323        }
1324
1325        if gate_type.starts_with("sky130_fd_sc_hd__udp_") {
1326            let udp = &udps[gate_type];
1327            let input_vals: Vec<bool> = gate.inputs.iter().map(|name| wires[name]).collect();
1328            let result = eval_udp_for_inputs(udp, &input_vals);
1329            wires.insert(gate.output.clone(), result);
1330            continue;
1331        }
1332
1333        let input_vals: Vec<bool> =
1334            gate.inputs.iter().map(|name| {
1335                *wires.get(name).unwrap_or_else(|| {
1336                    panic!(
1337                        "Wire '{}' not found in model '{}' at gate '{}' (type '{}'). Available wires: {:?}",
1338                        name, model.module_name, gate.output, gate_type,
1339                        wires.keys().collect::<Vec<_>>()
1340                    )
1341                })
1342            }).collect();
1343
1344        let result = match gate_type {
1345            "not" => !input_vals[0],
1346            "and" => input_vals.iter().all(|&v| v),
1347            "nand" => !input_vals.iter().all(|&v| v),
1348            "or" => input_vals.iter().any(|&v| v),
1349            "nor" => !input_vals.iter().any(|&v| v),
1350            "xor" => input_vals.iter().fold(false, |acc, &v| acc ^ v),
1351            "xnor" => !input_vals.iter().fold(false, |acc, &v| acc ^ v),
1352            _ => panic!("Unknown gate type: {}", gate_type),
1353        };
1354
1355        wires.insert(gate.output.clone(), result);
1356    }
1357
1358    wires[output_pin]
1359}

1436    /// Helper: set up CellInputs and a bool->aigpin mapping for testing.
1437    /// Returns (CellInputs, input_name_to_bool_map) for the given model.
1438    fn setup_test_inputs(
1439        model: &BehavioralModel,
1440        values: &[bool],
1441    ) -> (CellInputs, HashMap<String, bool>) {
1442        assert_eq!(model.inputs.len(), values.len());
1443        let mut cell_inputs = CellInputs::new();
1444        let mut bool_map = HashMap::new();
1445
1446        for (i, (name, &val)) in model.inputs.iter().zip(values.iter()).enumerate() {
1447            // Assign aigpin (i+1) to each input
1448            // aigpin_iv = ((i+1) << 1) | 0  for non-inverted
1449            let aigpin_iv = (i + 1) << 1;
1450            cell_inputs.set_pin(name, aigpin_iv);

1452        }

1457    /// Evaluate a DecompResult with concrete boolean inputs.
1458    /// `pin_values` maps aigpin (not aigpin_iv) to bool.
1459    fn eval_decomp_bool(decomp: &DecompResult, pin_values: &HashMap<usize, bool>) -> bool {
1460        let mut gate_outputs: Vec<bool> = Vec::new();
1461
1462        for (a_ref, b_ref) in &decomp.and_gates {
1463            let a_val = resolve_decomp_ref_bool(*a_ref, pin_values, &gate_outputs);
1464            let b_val = resolve_decomp_ref_bool(*b_ref, pin_values, &gate_outputs);
1465            gate_outputs.push(a_val && b_val);
1466        }
1467
1468        let output = if decomp.output_idx < 0 {
1469            let gate_idx = (-decomp.output_idx - 1) as usize;
1470            gate_outputs[gate_idx]
1471        } else {
1472            let aigpin = decomp.output_idx as usize;
1473            *pin_values
1474                .get(&aigpin)
1475                .unwrap_or_else(|| panic!("Pin {} not found in values map", aigpin))
1476        };
1477
1478        if decomp.output_inverted {
1479            !output
1480        } else {
1481            output
1482        }
1483    }
1484
1485    fn resolve_decomp_ref_bool(
1486        ref_val: i64,
1487        pin_values: &HashMap<usize, bool>,
1488        gate_outputs: &[bool],
1489    ) -> bool {
1490        if ref_val < 0 {
1491            let abs_ref = -ref_val;
1492            let gate_idx = ((abs_ref - 1) / 2) as usize;
1493            let inverted = (abs_ref % 2) == 0;
1494            let val = gate_outputs[gate_idx];
1495            if inverted {
1496                !val
1497            } else {
1498                val
1499            }
1500        } else {
1501            // ref_val is aigpin_iv: pin = ref_val >> 1, inverted = ref_val & 1
1502            let aigpin = (ref_val >> 1) as usize;
1503            let inverted = (ref_val & 1) != 0;
1504            let val = *pin_values.get(&aigpin).unwrap_or_else(|| {
1505                panic!(
1506                    "Pin {} not found in values map (ref_val={})",
1507                    aigpin, ref_val
1508                )
1509            });
1510            if inverted {
1511                !val
1512            } else {
1513                val
1514            }
1515        }
1516    }

1538    /// Check if all input pins of a model are recognized by CellInputs.
1539    fn all_inputs_supported(model: &BehavioralModel) -> bool {
1540        let supported = [
1541            "A", "A_N", "A0", "A1", "A1_N", "A2", "A2_N", "A3", "A4", "B", "B_N", "B1", "B1_N",
1542            "B2", "C", "C_N", "C1", "C2", "D", "D_N", "D1", "S", "S0", "S1", "CIN", "SET_B",
1543            "RESET_B", "SLEEP", "SLEEP_B",
1544        ];
1545        model
1546            .inputs
1547            .iter()
1548            .all(|name| supported.contains(&name.as_str()))
1549    }

1551    /// Exhaustive truth-table test for all combinational cells with supported pins.
1552    /// For each cell: evaluate all 2^N input combinations through both
1553    /// the AIG decomposition and the direct gate evaluator, verify they match.
1554    #[test]
1555    fn test_all_cells_vs_pdk() {
1556        let pdk = load_test_pdk();
1557
1558        let mut tested = 0;
1559        let mut skipped_inputs = 0;
1560        let mut skipped_size = 0;
1561
1562        // Sort keys for deterministic output
1563        let mut cell_types: Vec<&String> = pdk.models.keys().collect();
1564        cell_types.sort();
1565
1566        for cell_type in cell_types {
1567            let model = &pdk.models[cell_type];
1568            let num_inputs = model.inputs.len();
1569
1570            // Skip cells with unsupported input pin names
1571            if !all_inputs_supported(model) {
1572                skipped_inputs += 1;
1573                continue;
1574            }
1575
1576            // Skip cells with too many inputs for exhaustive testing (> 8 inputs)
1577            if num_inputs > 8 {
1578                skipped_size += 1;
1579                continue;
1580            }
1581
1582            let num_combos = 1u32 << num_inputs;
1583
1584            for output_pin in &model.outputs {
1585                for combo in 0..num_combos {
1586                    // Convert combo to input booleans
1587                    let input_bools: Vec<bool> =
1588                        (0..num_inputs).map(|i| ((combo >> i) & 1) != 0).collect();
1589
1590                    // Set up CellInputs
1591                    let (cell_inputs, bool_map) = setup_test_inputs(model, &input_bools);
1592
1593                    // Evaluate via direct gate interpretation (reference)
1594                    let expected = eval_behavioral_model(model, &bool_map, output_pin, &pdk.udps);
1595
1596                    // Evaluate via AIG decomposition
1597                    let decomp = decompose_with_pdk(cell_type, &cell_inputs, output_pin, &pdk);
1598
1599                    // Build pin_values map for decomp evaluator
1600                    let mut pin_values: HashMap<usize, bool> = HashMap::new();
1601                    pin_values.insert(0, false); // const-0
1602                    for (i, &val) in input_bools.iter().enumerate() {
1603                        pin_values.insert(i + 1, val);
1604                    }
1605
1606                    let actual = eval_decomp_bool(&decomp, &pin_values);
1607
1608                    assert_eq!(
1609                        actual, expected,
1610                        "Mismatch for cell '{}' output '{}' with inputs {:?}: AIG={}, expected={}",
1611                        cell_type, output_pin, input_bools, actual, expected
1612                    );
1613                }
1614            }
1615            tested += 1;
1616        }
1617
1618        println!(
1619            "Tested {} cell types exhaustively, skipped {} (unsupported pins), {} (too many inputs)",
1620            tested, skipped_inputs, skipped_size
1621        );
1622        assert!(
1623            tested > 50,
1624            "Expected to test at least 50 cell types, but only tested {}",
1625            tested
1626        );
1627    }
